#include <iostream>
#include "Windows.h"

using namespace std;

int main()
{
	setlocale(LC_ALL, "rus");

	DWORD size;
	char line[256];
	DWORD buf;
	DWORD dwFile;
	/*TCHAR hNameFile[40];
	TCHAR hNameFileToCopy[40];
	cout << "Введите название исходного файла: " << endl;
	wcin >> hNameFile;
	cout << "Введите название целевого файла: " << endl;
	wcin >> hNameFileToCopy;*/


	HANDLE hFile = CreateFile(/*hNameFile*/ TEXT("C:\\Фильмы\\text.txt"), GENERIC_READ, FILE_SHARE_READ , NULL, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
	if (hFile == INVALID_HANDLE_VALUE)
	{
		printf("Error open file\n");
		return -1;
	}

	HANDLE hFileToCopy = CreateFile(/*hNameFileToCopy*/ TEXT("C:\\Фильмы\\text1.txt"), GENERIC_WRITE, FILE_SHARE_WRITE, NULL, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
	if (hFileToCopy == INVALID_HANDLE_VALUE)
	{
		printf("Error open file\n");
		return -1;
	}

	size = GetFileSize(hFile, &size);
	ReadFile(hFile, line, size, &buf, NULL);
	WriteFile(hFileToCopy, line, size, &dwFile, NULL);
	
	CloseHandle(hFile);
	CloseHandle(hFileToCopy);

	cout << "Файл успешно скопирован!" << endl;

	return 0;
}
