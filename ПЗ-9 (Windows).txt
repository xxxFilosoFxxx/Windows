// ПЗ-9 (Windows).cpp : Этот файл содержит функцию "main". Здесь начинается и заканчивается выполнение программы.
//
#include <iostream>
#include "Windows.h"
#include <list>
#include <string>

using namespace std;

int main()
{
	setlocale(LC_ALL, "RUS");

	DWORD file_size;
	DWORD dwTemp, dwWritten;

	HANDLE hFile = CreateFile(TEXT("C:\\Фильмы\\text.txt"), GENERIC_WRITE | GENERIC_READ, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
	if (hFile == INVALID_HANDLE_VALUE)
	{
		printf("Error open file\n");
		return -1;
	}

	file_size = GetFileSize(hFile, NULL);
	char* text = new char[file_size];

	ReadFile(hFile, text, file_size, &dwTemp, NULL);
	if (file_size != dwTemp) {
		CloseHandle(hFile);
		return GetLastError();
	}

	string s_text; 
	string buf;
	list<string> my_list;

	for (int i = 0; i < file_size; i++)
	{
		s_text.push_back(text[i]);
	}

	for (int i = 0; i < s_text.size(); i++)
	{
		if (s_text[i] != '\n')
			buf += s_text[i];
		else
		{ 
			my_list.push_back(buf);
			buf.clear();
		}
		if (i == s_text.size() - 1)
			my_list.push_back(buf + '\r');
	}
	my_list.sort();
	my_list.unique();

	HANDLE s_hHeap;

	s_hHeap = HeapCreate(HEAP_NO_SERIALIZE, 0, 0);

	HANDLE hFileOut = CreateFile(TEXT("C:\\Фильмы\\result.txt"), GENERIC_WRITE | GENERIC_READ, 0, NULL, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
	if (hFileOut == INVALID_HANDLE_VALUE)
	{
		printf("Error open file\n");
		return -1;
	}

	for (auto it = my_list.begin(); it != my_list.end(); it++)
	{
		char* p = (char*)HeapAlloc(s_hHeap, 0, it->size());
		for (int j = 0; j < it->size(); j++)
			p[j] = (*it)[j];
		WriteFile(hFileOut, p, it->size(), &dwWritten, NULL);
		HeapFree(s_hHeap, 0, p);
	}

	HeapDestroy(s_hHeap);
	delete text;

	CloseHandle(hFile);
	CloseHandle(hFileOut);

	system("pause");
	return 0;
}
