#include <iostream>
#include "Windows.h"
#include "conio.h"

using namespace std;

int main()
{
	HKEY hKey;
	TCHAR nameValue[20];
	TCHAR text[20];
	char hPath[] = "Path\\key\\";

	setlocale(LC_ALL, "rus");
	bool done = true;
	while (done)
	{
		printf("0 - Exit\n1 - Write value\n-->");
		int ch = _getch();
		switch (ch) {
		case '0':
			done = false;
			break;
		case '1':
			cout << "Запросить имя ключа" << endl;
			cin >> nameValue;
			cout << "Запросить значение данных" << endl;
			cin >> text;

			if (RegCreateKeyEx(HKEY_CURRENT_USER, hPath, 0, NULL, REG_OPTION_VOLATILE, KEY_WRITE, NULL, &hKey, NULL) != ERROR_SUCCESS)
			{
				cout << "Ошибка создания файла!\n";
				return -1;
			}

			if (RegSetValueEx(hKey, nameValue, 0, REG_SZ, (BYTE*)text, sizeof(text)) != ERROR_SUCCESS)
			{
				cout << "Ошибка записи в файл!\n";
				return -1;
			}

			RegCloseKey(hKey);
			
			break;
		default:
			printf("Bad input!\n");
		}
	}


}
